Connect CrewAI, LangChain, or any Python agent to the /verify endpoint.

agent_passport_client.py 
"""Verify another agent's passport from CrewAI or LangChain."""
import httpx

PASSPORT_API = "{{ request.url.scheme }}://{{ request.url.netloc }}"

def verify_agent_passport(agent_id: str) -> dict:
    """
    Check if an agent is active, on trial or paid, and credit balance.
    Costs 1 prepaid credit from the target agent's wallet.
    Returns HTTP 402 if their balance is exhausted.
    """
    response = httpx.get(
        f"{PASSPORT_API}/verify/{agent_id}",
        headers={"Accept": "application/json"},
        timeout=30.0,
    )
    response.raise_for_status()
    return response.json()
